SELECT 
  NON EMPTY { [Measures].[Driving Lesson Count] } ON COLUMNS, 
  NON EMPTY { 
    FILTER(
      [Instructor].[PESEL Instructor].[PESEL Instructor].ALLMEMBERS, 
      [Instructor].[PESEL Instructor].CurrentMember.Name = "10060284857"
    ) * 
    [Instructor].[Name].[Name].ALLMEMBERS *
    [Instructor].[Surname].[Surname].ALLMEMBERS *
    [Instructor].[ID Instructor].[ID Instructor].ALLMEMBERS *
    [Instructor].[Is Current].[Is Current].ALLMEMBERS
  } 
  DIMENSION PROPERTIES MEMBER_CAPTION, MEMBER_UNIQUE_NAME ON ROWS 
FROM [HD Driving School]
CELL PROPERTIES VALUE, BACK_COLOR, FORE_COLOR, FORMATTED_VALUE, FORMAT_STRING, FONT_NAME, FONT_SIZE, FONT_FLAGS
